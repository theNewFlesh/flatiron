21 # type: (Filepath, Filepath, str) -> dict[str, str] 

22 ''' 

23 Creates directory structure for Tensorboard project.

30 Returns: 

31 dict: Project details. 

32 '''

58def enforce_callbacks(log_directory, checkpoint_pattern): 

59 # type: (Filepath, str) -> None 

60 ''' 

61 Enforces callback parameters. 

62 

63 Args: 

64 log_directory (str or Path): Tensorboard project log directory. 

65 checkpoint_pattern (str): Filepath pattern for checkpoint callback. 

66 

67 Raises: 

68 EnforceError: If log directory does not exist. 

69 EnforeError: If checkpoint pattern does not contain '{epoch}'. 

70 ''' 

71 log_dir = Path(log_directory) 

72 msg = f'Log directory: {log_dir} does not exist.' 

73 Enforce(log_dir.is_dir(), '==', True, message=msg) 

74 

75 match = re.search(r'\{epoch.*?\}', checkpoint_pattern) 

76 msg = "Checkpoint pattern must contain '{epoch}'. " 

77 msg += f'Given value: {checkpoint_pattern}' 

78 msg = msg.replace('{', '{{').replace('}', '}}') 

79 Enforce(match, '!=', None, message=msg)

83def pad_layer_name(name, length=18): 

84 # type: (str, int) -> str 

85 ''' 

86 Pads underscores in a given layer name to make the string achieve a given 

87 length. 

88 

89 Args: 

90 name (str): Layer name to be padded. 

91 length (int): Length of output string. Default: 18. 

92 

93 Returns: 

94 str: Padded layer name. 

95 ''' 

96 if length == 0: 

97 return name 

98 

99 if '_' not in name: 

100 name += '_' 

101 delta = length - len(re.sub('_', '', name)) 

102 return re.sub('_+', '_' * delta, name)

105def unindent(text, spaces=4): 

106 # type: (str, int) -> str 

107 ''' 

108 Unindents given block of text according to given number of spaces. 

109 

110 Args: 

111 text (str): Text block to unindent. 

112 spaces (int, optional): Number of spaces to remove. Default: 4. 

113 

114 Returns: 

115 str: Unindented text. 

116 ''' 

117 output = text.split('\n') # type: Any 

118 regex = re.compile('^ {' + str(spaces) + '}') 

119 output = [regex.sub('', x) for x in output] 

120 output = '\n'.join(output) 

121 return output

124def slack_it( 

125 title, # type: str 

126 channel, # type: str 

127 url, # type: str 

128 config=None, # type: Optional[dict] 

129 stopwatch=None, # type: Optional[StopWatch] 

130 timezone='UTC', # type: str 

131 suppress=False, # type: bool 

132): 

133 # type: (...) -> Union[str, HTTPResponse] 

134 ''' 

135 Compose a message from given arguments and post it to slack. 

136 

137 Args: 

138 title (str): Post title. 

139 channel (str): Slack channel. 

140 url (str): Slack URL. 

141 config (dict, optional): Parameter dict. Default: None. 

142 stopwatch (StopWatch, optional): StopWatch instance. Default: None. 

143 timezone (str, optional): Timezone. Default: UTC. 

144 suppress (bool, optional): Return message, rather than post it to Slack. 

145 Default: False. 

146 

147 Returns: 

148 HTTPResponse: Slack response. 

149 ''' 

150 now = datetime.now(tz=pytz.timezone(timezone)).isoformat() 

151 cfg = config or {} 

152 delta = 'none' 

153 hdelta = 'none' 

154 if stopwatch is not None: 

155 hdelta = stopwatch.human_readable_delta 

156 delta = str(stopwatch.delta) 

157 

158 config_ = yaml.safe_dump(cfg, indent=4) 

159 message = f''' 

160 {title.upper()} 

161 

162 RUN TIME: 

163 ```{hdelta} ({delta})``` 

164 POST TIME: 

165 ```{now}``` 

166 CONFIG: 

167 ```{config_}``` 

168 '''[1:-1] 

169 message = unindent(message, spaces=8) 

170 

171 if suppress: 

172 return message 

173 return lbt.post_to_slack(url, channel, message) # pragma: no cover

176def get_module_function(name, module): 

177 # type: (str, str) -> Callable[[Any], Any] 

178 ''' 

179 Get a function from a given module. 

180 

181 Args: 

182 name (str): Function name. 

183 module (str): Module name. 

184 

185 Raises: 

186 NotImplementedError: If function is not found in module. 

187 

188 Returns: 

189 function: Module function. 

190 ''' 

191 members = inspect.getmembers(sys.modules[module]) 

192 funcs = dict(filter(lambda x: inspect.isfunction(x[1]), members)) 

193 if name in funcs: 

194 return funcs[name] 

195 raise NotImplementedError(f'Function not found: {name}')
